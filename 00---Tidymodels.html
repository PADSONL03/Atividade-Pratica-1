  step_zv(all_predictors()) %>% # remove demais colunas que s tenham um nico valor
  step_mutate(
    Elevator = as.factor(ifelse(Elevator == 1, "Sim", "No")),
    Furnished = as.factor(ifelse(Furnished == 1, "Sim", "No")),
    Swimming.Pool = as.factor(ifelse(Swimming.Pool == 1, "Sim", "No")),
    New = as.factor(ifelse(New == 1, "Sim", "No"))
  ) %>% #transforma flags em nominais para posteriormente serem transformadas em dummys
  step_integer(c(Latitude,Longitude)) %>% # transforma latitude e longitude em infos numricas
